Get Started in 30 Seconds

# Install AetherPost
pip install aetherpost

# Super simple setup - just one command!
aetherpost init

# Create your first AI-powered campaign
cat > campaign.yaml << EOF
name: "AI App Launch"
concept: "Revolutionary AI-powered productivity tool"
platforms: [twitter, reddit, youtube, bluesky]
content:
  style: professional
  action: "Try it now!"
  ai_enhanced: true
EOF

# Generate AI content preview
aetherpost plan --ai

# Deploy when ready
aetherpost apply
# Clone the repository
git clone https://github.com/fununnn/aetherpost.git
cd aetherpost

# Install dependencies (includes OpenAI, YouTube, Reddit APIs)
pip install -r requirements.txt
pip install -e .

# Quick setup - simple!
aetherpost init
# Try the self-promotion feature!
# AetherPost promoting itself using its own capabilities

# Setup environment
cp .env.self_promotion .env
# Add your API keys to .env

# Run interactive self-promotion
./scripts/self_promote.sh

# Or automated self-promotion
python aetherpost_self_promotion.py

# See the meta magic happen! 🎭
